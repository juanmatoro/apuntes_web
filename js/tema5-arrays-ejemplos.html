2filter 

const tareas = [
  { id: 1, done: true },
  { id: 2, done: false },
  { id: 3, done: false }
];

const pendientes = tareas.filter(t => !t.done);

3reduce 

const carrito = [
  { producto: 'Teclado', precio: 30, cantidad: 2 },
  { producto: 'Ratón', precio: 20, cantidad: 1 }
];

const total = carrito.reduce((acc, item) => {
  return acc + item.precio * item.cantidad;
}, 0);
Uso real: total de carrito.

6findIndex 

const carrito = [
  { id: 1, cantidad: 2 },
  { id: 2, cantidad: 1 }
];

const idx = carrito.findIndex(item => item.id === 2);
if (idx !== -1) carrito[idx].cantidad += 1;
Uso real: actualizar una posición concreta.

7slice 

const resultados = [/* 100 elementos */];
const pagina = 2;
const porPagina = 10;

const inicio = (pagina - 1) * porPagina;
const fin = inicio + porPagina;
const vista = resultados.slice(inicio, fin);
