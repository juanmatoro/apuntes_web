Pureza Inmutabilidad Pipeline Compose

1Función pura vs impura 

// pura
const areaCirculo = (radio) => Math.PI * radio * radio;

// impura
let ultimoMensaje = '';
const guardarMensaje = (msg) => {
  ultimoMensaje = msg;
  return msg.length;
};

2Actualización inmutable 

const tareas = [
  { id: 1, texto: 'Estudiar', hecha: false },
  { id: 2, texto: 'Practicar', hecha: false }
];

const marcarHecha = (lista, id) =>
  lista.map((t) => (t.id === id ? { ...t, hecha: true } : t));

const nuevas = marcarHecha(tareas, 2);

3Pipeline declarativo 

const ventas = [
  { tienda: 'A', importe: 100 },
  { tienda: 'B', importe: 40 },
  { tienda: 'A', importe: 70 }
];

const totalA = ventas
  .filter((v) => v.tienda === 'A')
  .map((v) => v.importe)
  .reduce((acc, n) => acc + n, 0);

// totalA = 170

4Composición utilitaria 

const compose = (...fns) => (x) => fns.reduceRight((v, fn) => fn(v), x);

const trim = (s) => s.trim();
const lower = (s) => s.toLowerCase();
const shout = (s) => `${s}!`;

const normalizar = compose(shout, lower, trim);
normalizar('  HoLa  '); // 'hola!'
