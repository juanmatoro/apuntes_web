Clases Herencia Encapsulación Prototipos

1Clases e instancias 

class Usuario {
  constructor(nombre) {
    this.nombre = nombre;
  }

  saludar() {
    return `Hola, soy ${this.nombre}`;
  }
}

const ana = new Usuario('Ana');
ana.saludar(); // 'Hola, soy Ana'

2Herencia con extends 

class Empleado extends Usuario {
  constructor(nombre, puesto) {
    super(nombre);
    this.puesto = puesto;
  }

  presentarse() {
    return `${this.saludar()} y trabajo como ${this.puesto}`;
  }
}

const luis = new Empleado('Luis', 'Backend');
luis.presentarse(); // 'Hola, soy Luis y trabajo como Backend'

3Encapsulación con campos privados 

class Cuenta {
  #saldo = 0;

  ingresar(cantidad) {
    this.#saldo += cantidad;
  }

  get saldo() {
    return this.#saldo;
  }
}

const c = new Cuenta();
c.ingresar(100);
c.saldo;   // 100
// c.#saldo; // SyntaxError

4Prototipos en práctica

Cadena de prototipos

const animal = {
  respirar() {
    return 'inhala/exhala';
  }
};

const perro = Object.create(animal);
perro.ladrar = () => 'guau';

perro.ladrar();    // propio objeto
perro.respirar();  // heredado del prototipo (animal)

class por debajo usa prototipos

class Perro {
  hablar() {
    return 'guau';
  }
}

const rex = new Perro();
rex.hasOwnProperty('hablar'); // false: está en Perro.prototype
Object.getPrototypeOf(rex) === Perro.prototype; // true
