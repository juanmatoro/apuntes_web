1Notación de objetos

Código

const producto = {
  id: 101,
  nombre: 'Teclado',
  'precio-base': 29.99,
};

producto.nombre;         // notación punto
producto['precio-base']; // notación corchetes

const clave = 'id';
producto[clave];         // clave dinámica

Vista previa

producto.nombre -> 'Teclado'
producto['precio-base'] -> 29.99
producto[clave] -> 101

2this en métodos de objeto

Método tradicional

const usuario = {
  nombre: 'Ana',
  saludar() {
    return `Hola ${this.nombre}`;
  }
};

usuario.saludar(); // 'Hola Ana'

Arrow en objeto (error común)

const usuario = {
  nombre: 'Ana',
  saludar: () => `Hola ${this.nombre}`
};

// this no apunta al objeto usuario

3Control de contexto: bind, call y apply

bind para fijar this

const persona = {
  nombre: 'Luis',
  saludar() {
    return `Hola ${this.nombre}`;
  }
};

const fn = persona.saludar;
const saludoSeguro = fn.bind(persona);
saludoSeguro(); // 'Hola Luis'

call y apply

function presentar(ciudad, pais) {
  return `${this.nombre} desde ${ciudad}, ${pais}`;
}

const persona = { nombre: 'Elena' };

presentar.call(persona, 'Sevilla', 'ES');
presentar.apply(persona, ['Sevilla', 'ES']);

4Funciones asociadas a clases

Métodos de instancia + getter

class Carrito {
  #items = 0;

  agregar() {
    this.#items += 1;
  }

  get total() {
    return this.#items;
  }
}

const c = new Carrito();
c.agregar();
c.total; // 1

Método estático

class Usuario {
  constructor(nombre) {
    this.nombre = nombre;
  }

  static anonimo() {
    return new Usuario('Invitado');
  }
}

const u = Usuario.anonimo();
u.nombre; // 'Invitado'

5Patrones de construcción: literal, función constructora, factory y class

Literal

const producto = {
  id: 10,
  nombre: 'Ratón',
  precio: 19.99
};

producto.precio; // 19.99

Función constructora

function Producto(id, nombre, precio) {
  this.id = id;
  this.nombre = nombre;
  this.precio = precio;
}

const p = new Producto(10, 'Ratón', 19.99);
p.nombre; // 'Ratón'

Factory + closure

function crearCuenta(titular, inicial = 0) {
  let saldo = inicial; // privado

  return {
    titular,
    ingresar(cantidad) {
      saldo += cantidad;
    },
    getSaldo() {
      return saldo;
    }
  };
}

const cuenta = crearCuenta('Ana', 50);
cuenta.ingresar(25);
cuenta.getSaldo(); // 75

Class

class Cuenta {
  #saldo = 0;

  constructor(titular, inicial = 0) {
    this.titular = titular;
    this.#saldo = inicial;
  }

  ingresar(cantidad) {
    this.#saldo += cantidad;
  }

  get saldo() {
    return this.#saldo;
  }
}

const c = new Cuenta('Ana', 50);
c.ingresar(25);
c.saldo; // 75

6Métodos nativos de construcción de objetos

Object.create

const base = {
  tipo: 'usuario',
  saludar() {
    return `Hola ${this.nombre}`;
  }
};

const u = Object.create(base);
u.nombre = 'Ana';
u.saludar(); // 'Hola Ana'

Object.assign

const base = { id: 1, nombre: 'Ana' };
const extra = { activo: true };

const user = Object.assign({}, base, extra);
// { id: 1, nombre: 'Ana', activo: true }

Spread

const base = { id: 1, nombre: 'Ana' };
const user = { ...base, rol: 'admin' };

// copia superficial + extensión
// { id: 1, nombre: 'Ana', rol: 'admin' }

Object.fromEntries

const pares = [
  ['id', 7],
  ['email', 'ana@mail.com']
];

const user = Object.fromEntries(pares);
// { id: 7, email: 'ana@mail.com' }

7__proto__, prototype y Object

prototype en constructora

function Usuario(nombre) {
  this.nombre = nombre;
}

Usuario.prototype.saludar = function () {
  return `Hola ${this.nombre}`;
};

const u = new Usuario('Ana');
u.saludar(); // 'Hola Ana'

Cadena de prototipos

Object.getPrototypeOf(u) === Usuario.prototype; // true
u.__proto__ === Usuario.prototype;              // true

Object.getPrototypeOf(Usuario.prototype) === Object.prototype; // true
Object.getPrototypeOf(Object.prototype); // null

Object como constructor y utilidad base

const o1 = new Object(); // equivalente a {}
o1.id = 1;

Object.keys(o1);      // ['id']
Object.values(o1);    // [1]
Object.entries(o1);   // [['id', 1]]
