1Clases y objetos

class Usuario {
  constructor(nombre) {
    this.nombre = nombre;
  }
  saludar() {
    return `Hola, ${this.nombre}`;
  }
}

2Herencia

class Admin extends Usuario {
  borrarUsuario(id) {
    return `Usuario ${id} eliminado`;
  }
}

3Encapsulación

class Cuenta {
  #saldo = 0;
  depositar(monto) { this.#saldo += monto; }
  get saldo() { return this.#saldo; }
}

4Modelo prototipal

function Animal(nombre) {
  this.nombre = nombre;
}
Animal.prototype.hablar = function () { return `${this.nombre} hace ruido`; };
