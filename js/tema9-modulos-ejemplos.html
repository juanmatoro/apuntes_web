4Módulos + handlers del DOM

ui/todoView.js

export function bindTodoEvents(listEl, onDelete) {
  listEl.addEventListener('click', (event) => {
    const btn = event.target.closest('button[data-id]');
    if (!btn) return;
    onDelete(btn.dataset.id);
  });
}

main.js

import { bindTodoEvents } from './ui/todoView.js';
import { removeTodo } from './state/todos.js';

const listEl = document.querySelector('#todo-list');
bindTodoEvents(listEl, removeTodo);
