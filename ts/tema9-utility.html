Tipos mapeados Transformar objetos Funciones

1Utility types mas usados

interface Usuario {
  id: string;
  nombre: string;
  email?: string;
}

type UsuarioParcial = Partial<Usuario>;
type UsuarioCompleto = Required<Usuario>;

2Pick, Omit y Record

type UsuarioPublico = Pick<Usuario, "id" | "nombre">;
type UsuarioInterno = Omit<Usuario, "email">;

type Roles = "admin" | "editor" | "viewer";
type Permisos = Record<Roles, boolean>;

3Utilidades para funciones

function crearSesion(userId: string, remember: boolean) {
  return { token: "abc", userId, remember };
}

type ParamsSesion = Parameters<typeof crearSesion>;
type ResultadoSesion = ReturnType<typeof crearSesion>;

Estas utilidades reducen duplicacion de tipos y evitan desalineaciones.
