1Guards basicos

function printId(id: string | number): string {
  if (typeof id === "string") return id.toUpperCase();
  return id.toFixed(0);
}

TypeScript entiende guards como typeof, instanceof e in.

2Type predicates personalizados

type Perro = { ladrar: () => void };
type Gato = { maullar: () => void };

function esPerro(animal: Perro | Gato): animal is Perro {
  return "ladrar" in animal;
}

3Exhaustividad con never

type Estado = "idle" | "loading" | "error";

function pintar(estado: Estado) {
  switch (estado) {
    case "idle": return "Esperando";
    case "loading": return "Cargando";
    case "error": return "Fallo";
    default: {
      const _check: never = estado;
      return _check;
    }
  }
}
